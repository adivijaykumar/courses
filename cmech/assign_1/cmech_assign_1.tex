\begin{homeworkProblem}
\textbf{Part (a)}
\begin{flalign*}
V(x) = \alpha x^2/2 + \beta x^4/4\\
F(x) = -\pdv{V}{x} = -\alpha x - \beta x^3
\end{flalign*}
Including the damping term, we write the equation of motion as,
$$m\ddot{x} + \delta \dot{x} = -\alpha x - \beta x^3$$
$$m\ddot{x} + \delta \dot{x} + \alpha x + \beta x^3 = 0$$

The total energy of the system $E=T+V=m\dot{x}^2/2 + \alpha x^2/2 + \beta x^4/4$. Taking the time derivative, one gets
$$\dot{E} = m\ddot{x}\dot{x} + \alpha x \dot{x} + \beta x^3\dot{x}$$
Substituting from the equation of motion for $m\ddot{x}$,
$$\dot{E}=-(\delta \dot{x} + \alpha x + \beta x^3)\dot{x} + \alpha x \dot{x} + \beta x^3\dot{x}$$
$$\dot{E} = -\delta \dot{x}^2$$
Hence energy is dissipated from the system at a rate $\delta \dot{x}^2$.



\end{homeworkProblem}

\begin{homeworkProblem}
    Give an appropriate positive constant \(c\) such that \(f(n) \leq c \cdot
    g(n)\) for all \(n > 1\).

    \begin{enumerate}
        \item \(f(n) = n^2 + n + 1\), \(g(n) = 2n^3\)
        \item \(f(n) = n\sqrt{n} + n^2\), \(g(n) = n^2\)
        \item \(f(n) = n^2 - n + 1\), \(g(n) = n^2 / 2\)
    \end{enumerate}

    \textbf{Solution}

    We solve each solution algebraically to determine a possible constant
    \(c\).
    \\

    \textbf{Part One}

    \[
        \begin{split}
            n^2 + n + 1 &=
            \\
            &\leq n^2 + n^2 + n^2
            \\
            &= 3n^2
            \\
            &\leq c \cdot 2n^3
        \end{split}
    \]

    Thus a valid \(c\) could be when \(c = 2\).
    \\

    \textbf{Part Two}

    \[
        \begin{split}
            n^2 + n\sqrt{n} &=
            \\
            &= n^2 + n^{3/2}
            \\
            &\leq n^2 + n^{4/2}
            \\
            &= n^2 + n^2
            \\
            &= 2n^2
            \\
            &\leq c \cdot n^2
        \end{split}
    \]

    Thus a valid \(c\) is \(c = 2\).
    \\

    \textbf{Part Three}

    \[
        \begin{split}
            n^2 - n + 1 &=
            \\
            &\leq n^2
            \\
            &\leq c \cdot n^2/2
        \end{split}
    \]

    Thus a valid \(c\) is \(c = 2\).

\end{homeworkProblem}
