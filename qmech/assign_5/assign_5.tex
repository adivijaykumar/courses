\begin{homeworkProblem}[1]
	\textbf{Part (a)}\\
	We first note that,
	\begin{align*}
	\lambda e^{-t/\tau} \mel{n}{x^2}{m} &= \lambda e^{-t/\tau}  \mel{n}{\dfrac{(a_+ + a_-)^2}{2m\omega}}{m} \\
	&=\lambda e^{-t/\tau}  \mel{n}{\dfrac{a_+^2 + a_-^2 + a_+ a_- + a_- a_+}{2m\omega}}{m}\\
	&= \lambda e^{-t/\tau} \dfrac{1}{2 m \omega} \qty[\sqrt{(m+1)(m+2)} \delta_{n,m+2} + \sqrt{m(m-1)} \delta_{n, m-2} + (2m-1) \delta_{n,m}]
	\end{align*}
	As is evident from above, a state $ \ket{m} $ can transition into  $ \ket{m}, \ket{m+2}, \ket{m-2} $ and no other states under a potential with spatial dependence that goes as $ x^2 $. In general, the $ k $-th order coefficient $ c^k_n(t) $ will have $ k $ terms of the form $ \mel{.}{x^2}{.} $. If we start out with ground state $ \ket{0} $, the final state will have contributions from the following states order by order
	\begin{align*}
	\order{\lambda} &\rightarrow \ket{0}, \ket{2} \\
	\order{\lambda^2} &\rightarrow \ket{0}, \ket{2}, \ket{4} \\
	\order{\lambda^3} &\rightarrow \ket{0}, \ket{2}, \ket{4}, \ket{6} \\ 
	\therefore \order{\lambda^k} &\rightarrow \ket{0}, \ket{2}, \ket{4}, \ldots \ket{2k}
	\end{align*}
	Hence, we see that the $ \ket{n} $ as mentioned in the question should be such that $ n $ is even, and the leading order contribution to the probability will $ \sim (\lambda^{n/2})^2 \sim \lambda^n $.\\
	
	\textbf{Part (b)}\\
	As described above, upto $ \order{\lambda^2} $ in probability (ie upto $ \order{\lambda} $ in the coefficients), $ \ket{2} $ is the only excited state that can be reached. From $ (5.7.17) $ of Sakurai, we have the relations, (with $ \ket{i,t_0;t} = \sum c_n(t) \ket{n} $)
	\begin{equation*}
	c_n^0 (t) = \delta_{ni} \qq{,} c_n^1 (t) = -i \int_{t_0}^{t} e^{i \omega_{ni} t'} V_{ni}(t') dt'
	\end{equation*}
	Let's calculate $ c_n^1 (t) $,
	\begin{align*}
	c_n^1 (t) &=  -i \lambda \int_{0}^{t} e^{i n \omega t'} \mel{n}{ x^2}{0} e^{-t'/\tau} dt' \\
	&= \dfrac{-i \lambda}{2 m \omega} (\sqrt{2} \delta_{n,2} + \delta_{n,0}) \int_{0}^{t} e^{i n \omega t'}  e^{-t'/\tau} dt'\\
	c_n^1 (t) &= \dfrac{-i \lambda}{2 m \omega} (\sqrt{2} \delta_{n,2} + \delta_{n,0} )\dfrac{e^{i n \omega t}  e^{-t/\tau} - 1}{i n \omega - 1/\tau} \\
	\implies c_2^1 (t) &= \dfrac{-i \lambda }{\sqrt{2} m \omega}  \dfrac{e^{2 i \omega t}  e^{-t/\tau} - 1}{2 i \omega - 1/\tau} \implies  \abs{c_2^1 (t)}^2 = \dfrac{\lambda^2 }{{2} m^2 \omega^2}  \dfrac{ e^{-2t/\tau} + 1 - 2 e^{-t / \tau } \cos 2 \omega t}{4 \omega^2 + 1/\tau^2} 
	\end{align*}
	$ \abs{c_2^1}^2 $ is the required probability.
\end{homeworkProblem}








\begin{homeworkProblem}[2]
	
	We don't need to apply any perturbation theory in this problem, and it can be solved exactly. The Hamiltonian is $ H = \lambda S_1 \cdot S_2 = \lambda( S^2 - S_1^2 - S_2^2 ) $. We consider the action of the Hamiltonian on the singlet state $ \ket{0 0} = \dfrac{\ket{+ -} - \ket{- +}}{\sqrt{2}}$ and $ \ket{1 0 }  = \dfrac{\ket{+ -} + \ket{- +}}{\sqrt{2}} $. We know $ H \ket{00} = -3\lambda /4 \ket{00}  $ and $ S^2 \ket{10} = \lambda/4 \ket{10} $. Initially the system is in $ \ket{+-} = \dfrac{\ket{00} + \ket{10}}{\sqrt{2}} $. Then we know, by the usual rules of time-evolution,
	\begin{align*}
	\ket{\psi_f(t)} = e^{i H t} \ket{+-} &= \dfrac{e^{i\lambda t/4}}{\sqrt{2}} \ket{10} + \dfrac{e^{-i3\lambda t/4}}{\sqrt{2}} \ket{00}\\
	&= \qty(\dfrac{e^{i\lambda t/4}  +{e^{-i3\lambda t/4}}}{2}) \ket{+-} + \qty(\dfrac{e^{i\lambda t/4}  - {e^{-i3\lambda t/4}}}{2}) \ket{-+}\\
	\implies \abs{\ip{+-}{{\psi_f(t)}}}^2 &= \abs{\qty(\dfrac{e^{i\lambda t/4}  +{e^{-i3\lambda t/4}}}{2})}^2  = \dfrac{1 + \cos \lambda t}{2} =  P(\ket{+-})\\
	\implies \abs{\ip{-+}{{\psi_f(t)}}}^2 &= \abs{\qty(\dfrac{e^{i\lambda t/4}  - {e^{-i3\lambda t/4}}}{2})}^2  = \dfrac{1 - \cos \lambda t}{2} = P(\ket{-+})\\
	\implies \abs{\ip{++}{{\psi_f(t)}}}^2 &= 0 =  P(\ket{++})\\
	\implies \abs{\ip{--}{{\psi_f(t)}}}^2 &= 0 =  P(\ket{--})
	\end{align*}
	where $ P(\ket{}) $ denotes probability of initial state to be in state $ \ket{} $.
\end{homeworkProblem}










\begin{homeworkProblem}[3]
	\textbf{Part (a)}\\
	From $ (5.7.17) $ of Sakurai, we have the relations, (with $ \ket{i,t_0;t} = \sum c_n(t) \ket{n} $)
	\begin{equation*}
	c_n^0 (t) = \delta_{ni} \qq{,} c_n^1 (t) = -i \int_{t_0}^{t} e^{i \omega_{ni} t'} V_{ni}(t') dt' 
	\end{equation*}
	For our problem, we have $ V = \lambda \delta (x - vt) $. We insert $ 1 = \int dx \ket{x} \bra{x} $ such that $ V_{ni}(t) = \int V(t) u_i^* (x) u_n (x) dx $. We have initial state $ u_i (x) $ and final state $ u_f(x) $. Hence, we can write the above coefficients as,
	\begin{align*}
	c_f^1 (t) &= -i \lambda \int_{-\infty}^{\infty} dx \int_{0}^{t} dt' e^{i (E_i - E_f) t'} \delta(x - vt') u_i^* (x) u_f (x)  \\
	 &= -i \lambda \int_{-\infty}^{\infty} dx  e^{i (E_i - E_f) x/v}  u_i^* (x) u_f (x) 
	\end{align*}
	Hence the probability is just $ \abs{c_f^1}^2 $
	
	\textbf{Part (b)}\\
	We now write,
	\begin{equation*}
	\delta(x-vt) = \dfrac{1}{2 \pi v} \int_{- \infty}^{\infty} d\omega  e^{i\omega (x/v - t)}
	\end{equation*}
	\begin{align*}
	\therefore c_f^1 (t) &= -i \lambda \int_{-\infty}^{\infty} dx \int_{0}^{t} dt' e^{i (E_i - E_f) t'} \dfrac{1}{2 \pi v} \int_{- \infty}^{\infty} d\omega  e^{i\omega (x/v - t')} u_i^* (x) u_f (x)  \\
	&= \dfrac{-i \lambda}{2 \pi v} \int_{-\infty}^{\infty} dx \int_{0}^{t} dt' e^{i (E_{if} - \omega) t'}  \int_{- \infty}^{\infty} d\omega  e^{i\omega x/v} u_i^* (x) u_f (x)  \\
	&=  \dfrac{-i \lambda}{2 \pi v} \int_{-\infty}^{\infty} dx \int_{- \infty}^{\infty} d\omega \delta(E_{fi} - \omega)  e^{i\omega x/v } u_i^* (x) u_f (x)4
	\end{align*}
	Integrating the above will give us the same expression as that is Part (a). We notice that there is this $ \delta (E_{fi} - \omega) $ term, which basically ensures energy conservation.
\end{homeworkProblem}














\begin{homeworkProblem}[4]
	The ground state wavefunction for a hydrogen-like atom is given by,
	\begin{equation*}
	\ket{0_Z} = \dfrac{1}{\sqrt{\pi}} \qty(\dfrac{Z}{a_0})^{3/2} e^{-Zr/a_0}
	\end{equation*}
	We need to find $ \abs{\ip{0_2}{0_1}}^2 $,
	\begin{align*}
	\ip{0_2}{0_1} &= \dfrac{1}{\pi} \qty(\dfrac{2}{a_0^2})^{3/2} \int_{0}^{2\pi} d \phi \int_{0}^{\pi} -d(\cos\theta)  \int_{0}^{\infty} dr r^2 e^{-3r/a_0} \\
	&= \dfrac{1}{\pi} \qty(\dfrac{2}{a_0^2})^{3/2} (4 \pi) \qty(\dfrac{2 a_0^3}{27})\\
	\ip{0_2}{0_1} &= \sqrt{8} \dfrac{8}{27}\\
	\implies \abs{\ip{0_2}{0_1}}^2 &\approx 0.7
	\end{align*}
	So, the probability is close to $ 0.7 $.
\end{homeworkProblem}















\begin{homeworkProblem}[5]
	\textbf{Part (a)}\\
	We first note for $ \ket{\psi_I(t)} = \sum_{n} {c_n(t) \ket{\alpha_n}} $
	\begin{align*}
	i \pdv{\ket{\psi_I}}{t} &= i \pdv{(e^{iH_0t} \ket{\psi_S})}{t} \\
	&= i \qty[e^{iH_0t}\pdv{\ket{\psi_S}}{t}  + iH_0 e^{iH_0t} \ket{\psi_S}]\\
	&= -e^{iH_0t} (H_0 + V) \ket{\psi_S} - H_0 e^{iH_0t} \ket{\psi_S}\\
	&=e^{iH_0t}  V \ket{\psi_S}\\
	\end{align*}
	
	\begin{align*}
	i \pdv{\ket{\psi_I}}{t} &= V_I \ket{\psi_I}\\
	i \pdv{\ip{\alpha_n}{\psi_I}}{t} &=  \mel{\alpha_n}{V_I}{\psi_I}\\
	\dot{c_n} &= -i  \mel{\alpha_n}{V_I}{\psi_I}\\
	\dot{c_n} &= -i  \mel{\alpha_n}{V}{\alpha_m} e^{i{(E_n - E_m)t}} c_m
	\end{align*}
	So for the given problem, we have
	\begin{align*}
	\ket{\psi_I(t)} &=  {c_1(t) \ket{1} + c_2(t) e^{iEt} \ket{2}}	\\
	\dot{c_1} = -i V_{11} c_1 - i V_{12} e^{-iEt} c_2 = - i \gamma e^{i (\omega-E) t} c_2  &\qq{and} \dot{c_2} = -i V_{21}  e^{iEt}  c_1 - i V_{22}c_2 = -i \gamma e^{i(E-\omega)t} c_1
	\end{align*} 
	To solve the above equations, we make the substitution $  c_1 = b_1 e^{i\Delta t} $ and $c_2 = b_2 e^{-i \Delta t}  $, where $ 2 \Delta = \omega - E$. We then have the equations in terms of $ b $'s,
	\begin{equation*}
	i \dot{b_1} = \Delta b_1 + \gamma b_2 \qq{and} i \dot{b_2} = 
	\gamma b_1 - \Delta b_2
	\end{equation*}
	These are coupled equations, and we can solve these by making the substitution $ b_1 = A e^{i \Omega t} $ and $  b_2 = B e^{i \Omega t} $. We then have,
	\begin{align*}
	- A \Omega = \Delta A + \gamma B &\qq{and} - B \Omega = \gamma A - \Delta B \\
	\qq{For non-trivial solutions,} -\dfrac{\gamma}{\Delta + \Omega} &= \dfrac{\Delta - \Omega}{\gamma} \implies \Omega = \pm \sqrt{\gamma^2 + \Delta^2} = \pm \Omega_0\\
	\implies c_1 = A_1 e^{i(\Delta + \Omega_0)t} + A_2 e^{i(\Delta - \Omega_0)t} &\qq{and}  c_2 = B_1 e^{i(-\Delta + \Omega_0)t} + B_2 e^{i(-\Delta - \Omega_0)t}
	\end{align*}
	We are told that at $ t=0 $, the system is in state $ \ket{1} \implies c_1(0) = 1, c_2(0) = 0 \implies A_1 = 1 - A_2, B_1 = - B_2$. We also know that $ \dot{c}_2(0) = - i \gamma c_1 (0) $ and $ \dot{c}_1(0) = - i \gamma c_2 (0)  $ which means,
	\begin{align*}
	-i (1-A_1) \Omega_0 +i A_1 \Omega_0 +i \Delta =0 &\implies A_1 = \frac{\Omega_0 -\Delta }{2 \Omega_0 } \qq{and}  A_2 = -\frac{\Omega_0 -\Delta }{2 \Omega_0 }\\
	2 i B_1 \Omega_0 =-i \gamma &\implies B_1 = -\frac{\gamma }{2 \Omega_0 } \qq{and} B_2 = 1 + \frac{\gamma }{2 \Omega_0 } \\
	 \implies c_1 =  \frac{\Omega_0 -\Delta }{2 \Omega_0 } e^{i(\Delta + \Omega_0)t} -\frac{\Omega_0 -\Delta }{2 \Omega_0 } e^{i(\Delta - \Omega_0)t} &\qq{and}  c_2 = -\frac{\gamma }{2 \Omega_0 }  e^{i(-\Delta + \Omega_0)t} + \qty(1 + \frac{\gamma }{2 \Omega_0 }) e^{i(-\Delta - \Omega_0)t}
	\end{align*}
	where $ \Delta = \dfrac{\omega - E}{2} $ and $ \Omega_0 = \sqrt{\gamma^2 + \Delta^2} $\\. We calculate $ \abs{c_2(t)}^2 $ using Mathematica, and we get,
	\begin{equation*}
	 \abs{c_2(t)}^2 = \frac{\gamma ^2 \sin ^2(t \Omega_0 )}{\Omega_0 ^2} \qq{and}	\abs{c_1(t)}^2 = 1 - \frac{\gamma ^2 \sin ^2(t \Omega_0 )}{\Omega_0 ^2}
	 \end{equation*}
	\textbf{Part (b)}\\
	To prove to all orders in perturbation, we consider $ \omega = E \implies \Omega_0 = \gamma \implies \abs{c_2(t)}^2 = {\sin ^2(t \gamma)} $. From the Dyson series, we have,
	\begin{equation*}
	c_2^n (t) =  \bra{2} (-i)^n \int_{0}^{t} dt' \ldots \int_{0}^{t^{(n-1)}} dt^n V_I (t') \ldots V_I(t^{(n)}) \ket{1}
	\end{equation*}
	Inserting a complete set of state before each $ V_I $
	\begin{equation*}
	c_2^n (t) =  \bra{2} (-i)^n \int_{0}^{t} dt' \ldots \int_{0}^{t^{(n-1)}}  dt^n V_I (t') (\ket{1} \bra{1} + \ket{2} \bra{2}) \ldots  (\ket{1} \bra{1} + \ket{2} \bra{2}) V_I(t^{(n)}) \ket{1}
	\end{equation*}
	There is a constraint that the initial state should be $ \ket{1} $ and the final state should be $  \ket{2} $. Since, $  \ket{1} \rightarrow \ket{2} $ and $  \ket{2} \rightarrow \ket{1} $ at each application of $ V_I $, terms with even $ n $ will vanish. For terms with odd $ n $, let's first consider $ n=3 $
	\begin{align*}
	c_2^3 (t) &=   (-i)^3 \int_{0}^{t} dt' \int_{0}^{t'} dt'' \int_{0}^{t''} dt'''  \mel{2}{V_I(t')}{1} \mel{1}{V_I(t'')}{2} \mel{2}{V_I(t''')}{1} \\
	&=   (-i)^3 \int_{0}^{t} dt' \int_{0}^{t'} dt'' \int_{0}^{t''} dt'''  \mel{2}{V_I(t')}{1} \mel{1}{V_I(t'')}{2} \mel{2}{V_I(t''')}{1}\\
	&=   (-i\gamma)^3 \int_{0}^{t} dt' \int_{0}^{t'} dt'' \int_{0}^{t''} dt'''  e^{iEt'} \mel{2}{V(t')}{1} e^{iEt''} \mel{1}{V(t'')}{2} e^{iEt''} \mel{2}{V(t''')}{1} \\
	&=   (-i \gamma)^3 \int_{0}^{t} dt' \int_{0}^{t'} dt'' \int_{0}^{t''} dt'''  e^{iEt'} e^{-iEt'}  e^{iEt''}  e^{-iEt''}  e^{iEt''} e^{-iEt''} \\
	&=   (-i \gamma)^3 \dfrac{t^3}{3!} 
	\end{align*}
	We see that because $ \omega = E $, all the matrix elements become independent of $ t $, and we get a very simple answer for $ c_2^n = (-i \gamma)^n \dfrac{t^n}{n!}  $. Hence,
	\begin{equation*}
	c_2 (t) = \sum_{\qq{odd n}}(-i \gamma)^n \dfrac{t^n}{n!} = -i \sin t \gamma  \implies  \abs{c_2(t)}^2  = \sin^2 t \gamma
	\end{equation*}
	Hence the formula is verified.
\end{homeworkProblem}
